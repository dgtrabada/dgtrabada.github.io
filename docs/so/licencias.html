#include <stdio.h>
#include <stdbool.h>
int main() {
int c,d,n=100;
bool es_primo=false;
c=0;
while (c < n) {
c++;
es_primo=true;
for (d = 2; d < c; ++d) {
if (c % d == 0) {
es_primo = false;
break;
}
}
if (es_primo) {
printf("%d ", c);
}
}
printf("\n");
return 0;
}

Compilar significa traducir un código de programación a codigo ejecutable por la máquina

gcc primos.c

Ejecutar es la acción de iniciar la carga de un programa o de un archivo ejecutable.

./a.out
